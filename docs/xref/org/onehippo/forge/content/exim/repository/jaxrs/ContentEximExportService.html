87      @Path("/")
88      @Consumes(MediaType.MULTIPART_FORM_DATA)
89      @Produces(MediaType.APPLICATION_OCTET_STREAM)
90      @POST
91      public Response exportContentToZip(@Context SecurityContext securityContext, @Context HttpServletRequest request,
92              @Multipart(value = "batchSize", required = false) String batchSizeParam,
93              @Multipart(value = "throttle", required = false) String throttleParam,
94              @Multipart(value = "publishOnImport", required = false) String publishOnImportParam,
95              @Multipart(value = "dataUrlSizeThreshold", required = false) String dataUrlSizeThresholdParam,
96              @Multipart(value = "docbasePropNames", required = false) String docbasePropNamesParam,
97              @Multipart(value = "documentTags", required = false) String documentTagsParam,
98              @Multipart(value = "binaryTags", required = false) String binaryTagsParam,
99              @Multipart(value = "paramsJson", required = false) String paramsJsonParam,
100             @Multipart(value = "params", required = false) Attachment paramsAttachment) {
101 
102         Logger procLogger = log;
103 
104         File tempLogFile = null;
105         PrintStream tempLogOut = null;
106         File baseFolder = null;
107         Session session = null;
108         ExecutionParams params = new ExecutionParams();
109         ProcessStatus processStatus = null;
110 
111         try {
112             tempLogFile = File.createTempFile(TEMP_PREFIX, ".log");
113             tempLogOut = new PrintStream(new BufferedOutputStream(new FileOutputStream(tempLogFile)));
114             procLogger = createTeeLogger(log, tempLogOut);
115 
116             if (getProcessMonitor() != null) {
117                 processStatus = getProcessMonitor().startProcess();
118                 fillProcessStatusByRequestInfo(processStatus, securityContext, request);
119                 processStatus.setLogFile(tempLogFile);
120             }
121 
122             baseFolder = Files.createTempDirectory(TEMP_PREFIX).toFile();
123             procLogger.info("ContentEximService#exportContentToZip begins at {}.", baseFolder);
124 
125             if (paramsAttachment != null) {
126                 final String json = attachmentToString(paramsAttachment, "UTF-8");
127                 if (StringUtils.isNotBlank(json)) {
128                     params = getObjectMapper().readValue(json, ExecutionParams.class);
129                 }
130             } else {
131                 if (StringUtils.isNotBlank(paramsJsonParam)) {
132                     params = getObjectMapper().readValue(paramsJsonParam, ExecutionParams.class);
133                 }
134             }
135             overrideExecutionParamsByParameters(params, batchSizeParam, throttleParam, publishOnImportParam,
136                     dataUrlSizeThresholdParam, docbasePropNamesParam, documentTagsParam, binaryTagsParam);
137 
138             if (processStatus != null) {
139                 processStatus.setExecutionParams(params);
140             }
141 
142             session = createSession();
143             Result result = ResultItemSetCollector.collectItemsFromExecutionParams(session, params);
144             session.refresh(false);
145 
146             FileObject baseFolderObject = VFS.getManager().resolveFile(baseFolder.toURI());
147             FileObject attachmentsFolderObject = baseFolderObject.resolveFile(BINARY_ATTACHMENT_REL_PATH);
148 
149             DocumentManager documentManager = new WorkflowDocumentManagerImpl(session);
150 
151             final WorkflowDocumentVariantExportTask documentExportTask = new WorkflowDocumentVariantExportTask(
152                     documentManager);
153             documentExportTask.setLogger(log);
154             documentExportTask.setBinaryValueFileFolder(attachmentsFolderObject);
155             documentExportTask.setDataUrlSizeThreashold(params.getDataUrlSizeThreshold());
156 
157             final DefaultBinaryExportTask binaryExportTask = new DefaultBinaryExportTask(documentManager);
158             binaryExportTask.setLogger(log);
159             binaryExportTask.setBinaryValueFileFolder(attachmentsFolderObject);
160             binaryExportTask.setDataUrlSizeThreashold(params.getDataUrlSizeThreshold());
161 
162             int batchCount = 0;
163 
164             Set<String> referredNodePaths = new LinkedHashSet<>();
165 
166             try {
167                 documentExportTask.start();
168                 batchCount = exportDocuments(procLogger, processStatus, params, documentExportTask, result, batchCount,
169                         baseFolderObject, referredNodePaths);
170             } finally {
171                 documentExportTask.stop();
172             }
173 
174             if (!referredNodePaths.isEmpty()) {
175                 ResultItemSetCollector.fillResultItemsForNodePaths(session, referredNodePaths, true, null, result);
176                 session.refresh(false);
177             }
178 
179             try {
180                 binaryExportTask.start();
181                 batchCount = exportBinaries(procLogger, processStatus, params, binaryExportTask, result, batchCount,
182                         baseFolderObject);
183             } finally {
184                 binaryExportTask.stop();
185             }
186 
187             session.logout();
188             session = null;
189 
190             procLogger.info("ContentEximService#exportContentToZip ends.");
191 
192             tempLogOut.close();
193             tempLogOut = null;
194             procLogger = log;
195 
196             final String tempLogOutString = FileUtils.readFileToString(tempLogFile, "UTF-8");
197             final File zipBaseFolder = baseFolder;
198 
199             final StreamingOutput entity = new StreamingOutput() {
200                 @Override
201                 public void write(OutputStream output) throws IOException, WebApplicationException {
202                     ZipArchiveOutputStream zipOutput = null;
203                     try {
204                         zipOutput = new ZipArchiveOutputStream(output);
205                         zipOutput.setCreateUnicodeExtraFields(ZipArchiveOutputStream.UnicodeExtraFieldPolicy.ALWAYS);
206 
207                         ZipCompressUtils.addEntryToZip(EXIM_EXECUTION_LOG_REL_PATH, tempLogOutString, "UTF-8",
208                                 zipOutput);
209                         ZipCompressUtils.addEntryToZip(EXIM_SUMMARY_BINARIES_LOG_REL_PATH,
210                                 binaryExportTask.getSummary(), "UTF-8", zipOutput);
211                         ZipCompressUtils.addEntryToZip(EXIM_SUMMARY_DOCUMENTS_LOG_REL_PATH,
212                                 documentExportTask.getSummary(), "UTF-8", zipOutput);
213                         ZipCompressUtils.addFileEntriesInFolderToZip(zipBaseFolder, "", zipOutput);
214                     } finally {
215                         zipOutput.finish();
216                         IOUtils.closeQuietly(zipOutput);
217                         FileUtils.deleteDirectory(zipBaseFolder);
218                     }
219                 }
220             };
221 
222             String fileName = "exim-export-" + DateFormatUtils.format(Calendar.getInstance(), "yyyyMMdd-HHmmss")
223                     + ".zip";
224             return Response.ok().header("Content-Disposition", "attachment; filename=\"" + fileName + "\"")
225                     .entity(entity).build();
226         } catch (Exception e) {
227             procLogger.error("Failed to export content.", e);
228             if (baseFolder != null) {
229                 try {
230                     FileUtils.deleteDirectory(baseFolder);
231                 } catch (Exception ioe) {
232                     procLogger.error("Failed to delete the temporary folder at {}", baseFolder.getPath(), e);
233                 }
234             }
235             final String message = new StringBuilder().append(e.getMessage()).append("\r\n").toString();
236             return Response.serverError().entity(message).build();
237         } finally {
238             procLogger.info("ContentEximService#exportContentToZip finally ends.");
239 
240             if (getProcessMonitor() != null) {
241                 try {
242                     getProcessMonitor().stopProcess(processStatus);
243                 } catch (Exception e) {
244                     procLogger.error("Failed to stop process.", e);
245                 }
246             }
247 
248             if (session != null) {
249                 try {
250                     session.logout();
251                 } catch (Exception e) {
252                     procLogger.error("Failed to logout JCR session.", e);
253                 }
254             }
255 
256             if (tempLogOut != null) {
257                 IOUtils.closeQuietly(tempLogOut);
258             }
259 
260             if (tempLogFile != null) {
261                 try {
262                     tempLogFile.delete();
263                 } catch (Exception e) {
264                     log.error("Failed to delete temporary log file.", e);
265                 }
266             }
267         }
268     }

270     private int exportBinaries(Logger procLogger, ProcessStatus processStatus, ExecutionParams params,
271             DefaultBinaryExportTask exportTask, Result result, int batchCount, FileObject baseFolder) throws Exception {
272         final String baseFolderUrlPrefix = baseFolder.getURL().toString() + "/";
273         final AntPathMatcher pathMatcher = new AntPathMatcher();
274 
275         for (ResultItem item : result.getItems()) {
276             if (isStopRequested(baseFolder)) {
277                 procLogger.info("Stop requested by file at {}/{}", baseFolder.getName().getPath(),
278                         STOP_REQUEST_FILE_REL_PATH);
279                 break;
280             }
281 
282             ContentMigrationRecord record = null;
283 
284             try {
285                 String handlePath = item.getPath();
286 
287                 if (!isBinaryPathIncluded(pathMatcher, params, handlePath)) {
288                     continue;
289                 }
290 
291                 if (!HippoNodeUtils.isBinaryPath(handlePath)) {
292                     continue;
293                 }
294 
295                 if (!exportTask.getDocumentManager().getSession().nodeExists(handlePath)) {
296                     continue;
297                 }
298 
299                 Node handle = exportTask.getDocumentManager().getSession().getNode(handlePath);
300                 Node variant = HippoNodeUtils.getFirstVariantNode(handle);
301 
302                 if (variant == null) {
303                     continue;
304                 }
305 
306                 String variantPath = variant.getPath();
307                 record = exportTask.beginRecord(variant.getIdentifier(), variantPath);
308 
309                 ContentNode contentNode = exportTask.exportBinarySetToContentNode(variant);
310                 record.setProcessed(true);
311 
312                 ContentNodeUtils.replaceDocbasesByPaths(exportTask.getDocumentManager().getSession(), contentNode,
313                         ContentNodeUtils.MIRROR_DOCBASES_XPATH);
314 
315                 Set<String> docbasePropNames = params.getDocbasePropNames();
316                 if (CollectionUtils.isNotEmpty(docbasePropNames)) {
317                     for (String docbasePropName : docbasePropNames) {
318                         ContentNodeUtils.replaceDocbasePropertiesByPaths(exportTask.getDocumentManager().getSession(),
319                                 contentNode, "properties[@itemName='" + docbasePropName + "']");
320                     }
321                 }
322 
323                 ContentNodeUtils.removeUrlPrefixInJcrDataValues(contentNode, baseFolderUrlPrefix);
324 
325                 applyTagContentProperties(contentNode, params.getBinaryTags());
326 
327                 String relPath = StringUtils.removeStart(ContentPathUtils.removeIndexNotationInNodePath(variantPath),
328                         "/");
329                 FileObject file = baseFolder.resolveFile(relPath + ".json");
330                 record.setAttribute("file", file.getName().getPath());
331                 exportTask.writeContentNodeToJsonFile(contentNode, file);
332 
333                 procLogger.debug("Exported document from {} to {}.", handlePath, file.getName().getPath());
334                 record.setSucceeded(true);
335             } catch (Exception e) {
336                 procLogger.error("Failed to process record: {}", record, e);
337                 if (record != null) {
338                     record.setErrorMessage(e.toString());
339                 }
340             } finally {
341                 if (record != null) {
342                     exportTask.endRecord();
343                     result.incrementTotalBinaryCount();
344                     if (record.isSucceeded()) {
345                         result.incrementSucceededBinaryCount();
346                     } else {
347                         result.incrementFailedBinaryCount();
348                     }
349                     if (processStatus != null) {
350                         processStatus.setProgress(result.getProgress());
351                     }
352                 }
353                 ++batchCount;
354                 if (batchCount % params.getBatchSize() == 0) {
355                     exportTask.getDocumentManager().getSession().refresh(false);
356                     if (params.getThrottle() > 0) {
357                         Thread.sleep(params.getThrottle());
358                     }
359                 }
360             }
361         }
362 
363         exportTask.getDocumentManager().getSession().refresh(false);
364 
365         return batchCount;
366     }

368     private int exportDocuments(Logger procLogger, ProcessStatus processStatus, ExecutionParams params,
369             WorkflowDocumentVariantExportTask exportTask, Result result, int batchCount, FileObject baseFolder,
370             Set<String> referredBinaryPaths) throws Exception {
371         final String baseFolderUrlPrefix = baseFolder.getURL().toString() + "/";
372         final AntPathMatcher pathMatcher = new AntPathMatcher();
373 
374         for (ResultItem item : result.getItems()) {
375             if (isStopRequested(baseFolder)) {
376                 procLogger.info("Stop requested by file at {}/{}", baseFolder.getName().getPath(),
377                         STOP_REQUEST_FILE_REL_PATH);
378                 break;
379             }
380 
381             ContentMigrationRecord record = null;
382 
383             try {
384                 String handlePath = item.getPath();
385 
386                 if (!isDocumentPathIncluded(pathMatcher, params, handlePath)) {
387                     continue;
388                 }
389 
390                 if (!HippoNodeUtils.isDocumentPath(handlePath)) {
391                     continue;
392                 }
393 
394                 if (!exportTask.getDocumentManager().getSession().nodeExists(handlePath)) {
395                     continue;
396                 }
397 
398                 Node handle = exportTask.getDocumentManager().getSession().getNode(handlePath);
399                 Map<String, Node> variantsMap = HippoNodeUtils.getDocumentVariantsMap(handle);
400                 Node variant = variantsMap.get(HippoStdNodeType.PUBLISHED);
401                 if (variant == null) {
402                     variant = variantsMap.get(HippoStdNodeType.UNPUBLISHED);
403                 }
404 
405                 if (variant == null) {
406                     continue;
407                 }
408 
409                 String variantPath = variant.getPath();
410                 record = exportTask.beginRecord(variant.getIdentifier(), variantPath);
411 
412                 Document document = new Document(variant.getIdentifier());
413                 ContentNode contentNode = exportTask.exportVariantToContentNode(document);
414                 record.setProcessed(true);
415 
416                 ContentNodeUtils.replaceDocbasesByPaths(exportTask.getDocumentManager().getSession(), contentNode,
417                         ContentNodeUtils.MIRROR_DOCBASES_XPATH, referredBinaryPaths);
418 
419                 Set<String> docbasePropNames = params.getDocbasePropNames();
420                 if (CollectionUtils.isNotEmpty(docbasePropNames)) {
421                     for (String docbasePropName : docbasePropNames) {
422                         ContentNodeUtils.replaceDocbasePropertiesByPaths(exportTask.getDocumentManager().getSession(),
423                                 contentNode, "properties[@itemName='" + docbasePropName + "']");
424                     }
425                 }
426 
427                 ContentNodeUtils.removeUrlPrefixInJcrDataValues(contentNode, baseFolderUrlPrefix);
428 
429                 applyTagContentProperties(contentNode, params.getDocumentTags());
430 
431                 String relPath = StringUtils.removeStart(ContentPathUtils.removeIndexNotationInNodePath(variantPath),
432                         "/");
433                 FileObject file = baseFolder.resolveFile(relPath + ".json");
434                 record.setAttribute("file", file.getName().getPath());
435 
436                 exportTask.writeContentNodeToJsonFile(contentNode, file);
437                 procLogger.debug("Exported document from {} to {}.", handlePath, file.getName().getPath());
438                 record.setSucceeded(true);
439             } catch (Exception e) {
440                 procLogger.error("Failed to process record: {}", record, e);
441                 if (record != null) {
442                     record.setErrorMessage(e.toString());
443                 }
444             } finally {
445                 if (record != null) {
446                     exportTask.endRecord();
447                     result.incrementTotalDocumentCount();
448                     if (record.isSucceeded()) {
449                         result.incrementSucceededDocumentCount();
450                     } else {
451                         result.incrementFailedDocumentCount();
452                     }
453                     if (processStatus != null) {
454                         processStatus.setProgress(result.getProgress());
455                     }
456                 }
457                 ++batchCount;
458                 if (batchCount % params.getBatchSize() == 0) {
459                     exportTask.getDocumentManager().getSession().refresh(false);
460                     if (params.getThrottle() > 0) {
461                         Thread.sleep(params.getThrottle());
462                     }
463                 }
464             }
465         }
466 
467         exportTask.getDocumentManager().getSession().refresh(false);
468 
469         return batchCount;
470     }
