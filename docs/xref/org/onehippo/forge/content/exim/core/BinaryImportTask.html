16  package org.onehippo.forge.content.exim.core;
17  
18  import org.onehippo.forge.content.pojo.model.ContentNode;
19  
20  /**
21   * <code>ContentMigrationTask</code> to import binary (gallery/asset) data from {@link ContentNode} objects.
22   */
23  public interface BinaryImportTask extends ContentMigrationTask {
24  
25      String createOrUpdateBinaryFolder(String folderLocation, String primaryTypeName, String [] folderTypes,
26              String [] galleryTypes) throws ContentMigrationException;
27  
28      /**
29       * Creates an imageset (e.g, <code>hippogallery:imageset</code>) content or
30       * an assetset (e.g, <code>hippogallery:exampleAssetSet</code>) content from a {@link ContentNode} object.
31       * @param contentNode a {@link ContentNode} object containing an imageset (e.g, <code>hippogallery:imageset</code>) content
32       *                    or an assetset (e.g, <code>hippogallery:exampleAssetSet</code>) content
33       * @param primaryTypeName primary node type of binary content.
34       *                        e.g, <code>hippogallery:imageset</code>, <code>hippogallery:exampleAssetSet</code>, etc.
35       * @param folderPath folder node path where this binary content should be created or updated
36       * @param name binary content node name
37       * @return created or updated binary handle node path
38       * @throws ContentMigrationException if importing fails
39       */
40      String createOrUpdateBinaryFromContentNode(ContentNode contentNode, String primaryTypeName, String folderPath,
41              String name) throws ContentMigrationException;
42  
43  }
