16  package org.onehippo.forge.content.exim.core.util;
17  
18  import org.apache.commons.vfs2.FileDepthSelector;
19  import org.apache.commons.vfs2.FileFilter;
20  import org.apache.commons.vfs2.FileSelectInfo;
21  import org.apache.commons.vfs2.FileSelector;
22  import org.apache.commons.vfs2.FileSystemException;
23  import org.apache.commons.vfs2.util.Messages;
24  
25  /**
26   * A {@link FileSelector} that selects all files in a particular depth range,
27   * with filtering a file by the given {@link FileFilter} instance.
28   */
29  public class FileFilterDepthSelector extends FileDepthSelector {
30  
31      /**
32       * The FileFilter.
33       */
34      private final FileFilter fileFilter;
35  
36      public FileFilterDepthSelector(final FileFilter fileFilter, int minDepth, int maxDepth) {
37          super(minDepth, maxDepth);
38          this.fileFilter = fileFilter;
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      public boolean includeFile(final FileSelectInfo fileInfo) throws Exception {
46          if (!super.includeFile(fileInfo)) {
47              return false;
48          }
49  
50          return accept(fileInfo);
51      }
52  
53      /**
54       * Check if {@code fileInfo} is acceptable by using the internal filter.
55       * @param fileInfo fileInfo
56       * @return true if {@code fileInfo} is acceptable by using the internal filter
57       */
58      protected boolean accept(final FileSelectInfo fileInfo) throws FileSystemException {
59          if (fileFilter != null) {
60              return fileFilter.accept(fileInfo);
61          }
62  
63          throw new IllegalArgumentException(Messages.getString("vfs.selectors/filefilter.missing.error"));
64      }
65  }
