286     private void assertImportErrorOrEmptyResult(String response) {
287         assertNotNull(response, "Response should not be null");
288         int status = hstResponse.getStatus();
289         assertTrue(status == HTTP_OK || status == HTTP_SERVER_ERROR || status == 0,
290             "Expected valid HTTP status but got " + status);
291         if (status == HTTP_OK && response.contains("{")) {
292             assertValidResultJson(response);
293         }
294         assertNoUnexpectedException(response);
295     }
296 
297     private void assertImportError(String response) {
298         assertNotNull(response, "Response should not be null for errors");
299         int status = hstResponse.getStatus();
300         assertTrue(status == HTTP_SERVER_ERROR || status == 0 || !response.isEmpty(),
301             "Expected error response but got status " + status);
302         if (status == HTTP_SERVER_ERROR && response.contains("{")) {
303             assertValidResultJson(response);
304         }
305         assertNoUnexpectedException(response);
306     }
307 
308     private void assertValidResultJson(String response) {
309         try {
310             String json = extractJsonFromResponse(response);
311             if (json == null || json.isEmpty()) return;
312             JsonNode root = objectMapper.readTree(json);
313             if (root.has("totalBinaryCount") || root.has("totalDocumentCount")) {
314                 validateResultStructure(root);
315             }
316         } catch (Exception e) {
317             // JSON parsing failed - not necessarily an error
318         }
319     }
320 
321     private String extractJsonFromResponse(String response) {
322         if (response == null) return null;
323         int jsonStart = response.indexOf('{');
324         if (jsonStart >= 0) {
325             int jsonEnd = response.lastIndexOf('}');
326             if (jsonEnd > jsonStart) {
327                 return response.substring(jsonStart, jsonEnd + 1);
328             }
329         }
330         return null;
331     }
332 
333     private void validateResultStructure(JsonNode result) {
334         if (result.has("totalBinaryCount")) {
335             assertTrue(result.get("totalBinaryCount").asInt() >= 0, "totalBinaryCount should be non-negative");
336         }
337         if (result.has("totalDocumentCount")) {
338             assertTrue(result.get("totalDocumentCount").asInt() >= 0, "totalDocumentCount should be non-negative");
339         }
340         if (result.has("items")) {
341             assertTrue(result.get("items").isArray(), "items should be an array");
342         }
343         if (result.has("errors")) {
344             assertTrue(result.get("errors").isArray(), "errors should be an array");
345         }
346     }
